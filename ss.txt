-- ========== ФУНКЦІЯ ФЛІКУ (на 100 футів у випадкову сторону) ==========
local function FlickPlayerAway(target)
    if not target or not target.Character then return end
    
    local targetRoot = target.Character:FindFirstChild("Humanoid") and target.Character.Humanoid.RootPart
    if not targetRoot then return end
    
    -- Випадковий кут (0-360 градусів)
    local randomAngle = math.random() * math.pi * 2
    -- 100 футів = ~30.48 студій (1 фут = 0.3048 студій)
    local distance = 100 * 0.3048  -- ~30.48 студій
    local randomDirection = Vector3.new(math.cos(randomAngle), 0, math.sin(randomAngle)) * distance
    local randomHeight = math.random(20, 50)  -- Додаткова висота
    
    -- Отримуємо позицію гравця
    local targetPos = targetRoot.Position
    local newPos = targetPos + randomDirection + Vector3.new(0, randomHeight, 0)
    
    -- Телепортуємо гравця
    targetRoot.CFrame = CFrame.new(newPos)
    targetRoot.Velocity = Vector3.new(0, 0, 0)
    targetRoot.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
    
    print(string.format("Flicked %s %.1f studs away (%.1f feet)", target.Name, distance, distance / 0.3048))
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "FLICK",
        Text = string.format("%s flung %.0f feet away!", target.Name, distance / 0.3048),
        Duration = 2
    })
end

-- ========== AUTOSTEAL (механіка з першого скрипта - ти крадеш у них) ==========
-- Це функція яка змушує гравця слідувати за тобою (ти "крадеш" у нього)
local function StartAutoSteal(target)
    if AutoStealConnection then
        AutoStealConnection:Disconnect()
        AutoStealConnection = nil
    end
    
    if not target or not target.Character then return false end
    
    AutoStealEnabled = true
    
    AutoStealConnection = RunService.Heartbeat:Connect(function()
        if not AutoStealEnabled or not target or not target.Character then
            if AutoStealConnection then AutoStealConnection:Disconnect() end
            AutoStealConnection = nil
            return
        end
        
        local targetChar = target.Character
        local targetHumanoid = targetChar:FindFirstChildOfClass("Humanoid")
        local targetRoot = targetHumanoid and targetHumanoid.RootPart
        local myChar = Player.Character
        local myRoot = myChar and myChar:FindFirstChildOfClass("Humanoid") and myChar:FindFirstChildOfClass("Humanoid").RootPart
        
        if targetRoot and myRoot then
            -- ГРАВЕЦЬ СЛІДУЄ ЗА ТОБОЮ (ти крадеш у нього)
            targetRoot.CFrame = myRoot.CFrame * CFrame.new(0, 0, 2)
            targetRoot.Velocity = myRoot.Velocity
            
            -- Блокуємо можливість сісти
            if targetHumanoid then
                targetHumanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
                targetHumanoid.AutoRotate = false
            end
        end
    end)
    
    return true
end

local function StopAutoSteal()
    AutoStealEnabled = false
    if AutoStealConnection then
        AutoStealConnection:Disconnect()
        AutoStealConnection = nil
    end
    
    -- Відновлюємо налаштування гравця
    if SelectedTarget and SelectedTarget.Character then
        local humanoid = SelectedTarget.Character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
            humanoid.AutoRotate = true
        end
    end
end

-- Оновлення списку гравців
local function UpdatePlayerList()
    for _, child in pairs(PlayerListFrame:GetChildren()) do
        if child:IsA("Frame") then child:Destroy() end
    end
    
    local yPos = 5
    local playersList = {}
    
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= Player then
            table.insert(playersList, plr)
        end
    end
    
    table.sort(playersList, function(a, b) return a.Name:lower() < b.Name:lower() end)
    
    for _, plr in ipairs(playersList) do
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(1, -10, 0, 30)
        btn.Position = UDim2.new(0, 5, 0, yPos)
        btn.BackgroundColor3 = Color3.fromRGB(50, 50, 55)
        btn.Text = plr.Name
        btn.TextColor3 = Color3.fromRGB(255, 255, 255)
        btn.Font = Enum.Font.Gotham
        btn.TextSize = 12
        btn.Parent = PlayerListFrame
        
        local btnCorner = Instance.new("UICorner")
        btnCorner.CornerRadius = UDim.new(0, 5)
        btnCorner.Parent = btn
        
        btn.MouseButton1Click:Connect(function()
            SelectedTarget = plr
            TargetDropdown.Text = plr.Name
            PlayerListFrame.Visible = false
            PlayerListFrame.Size = UDim2.new(1, -20, 0, 0)
        end)
        
        yPos = yPos + 35
    end
    
    PlayerListFrame.CanvasSize = UDim2.new(0, 0, 0, yPos + 5)
end

-- Показати/сховати список
TargetDropdown.MouseButton1Click:Connect(function()
    if PlayerListFrame.Visible then
        PlayerListFrame.Visible = false
        PlayerListFrame.Size = UDim2.new(1, -20, 0, 0)
    else
        UpdatePlayerList()
        PlayerListFrame.Visible = true
        PlayerListFrame.Size = UDim2.new(1, -20, 0, 120)
    end
end)

-- ========== ЛОГІКА ==========
AutoStealCheckbox.MouseButton1Click:Connect(function()
    if AutoStealEnabled then
        StopAutoSteal()
        Checkmark.Visible = false
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "AutoSteal",
            Text = "Stopped",
            Duration = 1
        })
    else
        if not SelectedTarget then
            game:GetService("StarterGui"):SetCore("SendNotification", {
                Title = "Error",
                Text = "Select a target first!",
                Duration = 2
            })
            return
        end
        StartAutoSteal(SelectedTarget)
        Checkmark.Visible = true
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "AutoSteal",
            Text = "Stealing from " .. SelectedTarget.Name,
            Duration = 2
        })
    end
end)

FlickButton.MouseButton1Click:Connect(function()
    if not SelectedTarget then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Error",
            Text = "Select a target first!",
            Duration = 2
        })
        return
    end
    
    -- Якщо автостил включений - вимикаємо на час фліку
    local wasEnabled = AutoStealEnabled
    if wasEnabled then
        StopAutoSteal()
        Checkmark.Visible = false
    end
    
    -- Флікаємо гравця
    FlickPlayerAway(SelectedTarget)
    
    -- Затримка
    task.wait(0.5)
    
    -- Вмикаємо знову
    if wasEnabled then
        StartAutoSteal(SelectedTarget)
        Checkmark.Visible = true
    end
end)

-- Оновлення при видаленні гравця

    if SelectedTarget == plr then
        SelectedTarget = nil
        TargetDropdown.Text = "Click to select player"
        if AutoStealEnabled then
            StopAutoSteal()
            Checkmark.Visible = false
        end
    end
end)

        StopAutoSteal()

        StartAutoSteal(SelectedTarget)
    end
end)
