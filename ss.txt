-- Функція оновлення списку гравців
local function RefreshPlayerList()
    for _, child in pairs(PlayerListFrame:GetChildren()) do
        if child:IsA("Frame") then child:Destroy() end
    end
    
    local yPos = 5
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= Player then
            local entry = Instance.new("Frame")
            entry.Size = UDim2.new(1, -10, 0, 35)
            entry.Position = UDim2.new(0, 5, 0, yPos)

            entry.Parent = PlayerListFrame
            
            local entryCorner = Instance.new("UICorner")
            entryCorner.CornerRadius = UDim.new(0, 5)
            entryCorner.Parent = entry
            
            local checkbox = Instance.new("TextButton")
            checkbox.Size = UDim2.new(0, 22, 0, 22)
            checkbox.Position = UDim2.new(0, 8, 0.5, -11)
            checkbox.BackgroundColor3 = Color3.fromRGB(70, 70, 75)
            checkbox.BorderSizePixel = 0
            checkbox.Text = ""
            checkbox.Parent = entry
            
            local checkCorner = Instance.new("UICorner")
            checkCorner.Parent = checkbox
            
            local checkmark = Instance.new("TextLabel")
            checkmark.Size = UDim2.new(1, 0, 1, 0)
            checkmark.BackgroundTransparency = 1
            checkmark.Text = "✓"
            checkmark.TextColor3 = Color3.fromRGB(0, 255, 0)
            checkmark.TextSize = 14
            checkmark.Font = Enum.Font.GothamBold
            checkmark.Visible = SelectedTargets[plr.Name] ~= nil
            checkmark.Parent = checkbox
            
            local nameLabel = Instance.new("TextLabel")
            nameLabel.Size = UDim2.new(1, -40, 1, 0)
            nameLabel.Position = UDim2.new(0, 38, 0, 0)
            nameLabel.BackgroundTransparency = 1
            nameLabel.Text = plr.Name
            nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
            nameLabel.TextSize = 13
            nameLabel.Font = Enum.Font.Gotham
            nameLabel.TextXAlignment = Enum.TextXAlignment.Left
            nameLabel.Parent = entry
            
            local speedText = Instance.new("TextLabel")
            speedText.Size = UDim2.new(0, 45, 1, 0)
            speedText.Position = UDim2.new(1, -50, 0, 0)
            speedText.BackgroundTransparency = 1
            speedText.Text = string.format("%.1fx", SpeedMultipliers[plr.Name] or 1)
            speedText.TextColor3 = Color3.fromRGB(100, 200, 150)
            speedText.TextSize = 11
            speedText.Font = Enum.Font.Gotham
            speedText.TextXAlignment = Enum.TextXAlignment.Right
            speedText.Parent = entry
            
            -- Виділення гравця
            local click = Instance.new("TextButton")
            click.Size = UDim2.new(1, 0, 1, 0)
            click.BackgroundTransparency = 1
            click.Text = ""
            click.Parent = entry
            
            click.MouseButton1Click:Connect(function()
                if SelectedTargets[plr.Name] then
                    SelectedTargets[plr.Name] = nil
                    checkmark.Visible = false
                else
                    SelectedTargets[plr.Name] = plr
                    checkmark.Visible = true
                end
            end)
            
            yPos = yPos + 42
        end
    end
    
    PlayerListFrame.CanvasSize = UDim2.new(0, 0, 0, yPos + 5)
end

-- Функція автостилу
local function AutoStylePlayer(target)
    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not AutoStyleActive or not target.Character then
            if connection then connection:Disconnect() end
            return
        end
        
        local targetChar = target.Character
        local targetRoot = targetChar:FindFirstChild("Humanoid") and targetChar.Humanoid.RootPart
        local playerChar = Player.Character
        local playerRoot = playerChar and playerChar:FindFirstChild("Humanoid") and playerChar.Humanoid.RootPart
        
        if targetRoot and playerRoot then
            -- Застосування швидкості
            local humanoid = targetChar:FindFirstChildOfClass("Humanoid")
            if humanoid then
                local speed = (SpeedMultipliers[target.Name] or 1) * 16
                humanoid.WalkSpeed = speed
            end
            
            -- Слідування за гравцем
            targetRoot.CFrame = playerRoot.CFrame * CFrame.new(0, 0, 2)
            targetRoot.Velocity = playerRoot.Velocity
        end
    end)
    return connection
end

-- Запуск автостилу
local function StartAutoStyle()
    if AutoStyleActive then return end
    
    for name, conn in pairs(Connections) do
        if conn then conn:Disconnect() end
    end
    Connections = {}
    
    local count = 0
    for name, plr in pairs(SelectedTargets) do
        if plr and plr.Character then
            Connections[name] = AutoStylePlayer(plr)
            count = count + 1
        end
    end
    
    if count > 0 then
        AutoStyleActive = true
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "AutoStyle",
            Text = "Started on " .. count .. " player(s)",
            Duration = 2
        })
    end
end

-- Зупинка автостилу
local function StopAutoStyle()
    for name, conn in pairs(Connections) do
        if conn then conn:Disconnect() end
    end
    Connections = {}
    AutoStyleActive = false
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "AutoStyle",
        Text = "Stopped",
        Duration = 2
    })
end

-- Функція фліку (відкидання гравця)
local function FlickPlayer(target)
    if not target.Character then return end
    
    local root = target.Character:FindFirstChild("Humanoid") and target.Character.Humanoid.RootPart
    if root then
        -- Зупиняємо автостил
        local wasActive = AutoStyleActive
        if wasActive then StopAutoStyle() end
        
        -- Відкидаємо
        root.Velocity = Vector3.new(0, 200, 0)
        root.AssemblyLinearVelocity = Vector3.new(0, 250, 0)
        
        -- Чекаємо
        task.wait(0.5)
        
        -- Запускаємо автостил знову
        if wasActive then StartAutoStyle() end
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Flick",
            Text = "Flicked " .. target.Name,
            Duration = 2
        })
    end
end

-- Оновлення швидкості для всіх
local function UpdateGlobalSpeed(value)
    for name, _ in pairs(SelectedTargets) do
        SpeedMultipliers[name] = value
    end
    RefreshPlayerList()
end

-- Повзунок швидкості
local dragging = false
SpeedSlider.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
    end
end)

SpeedSlider.InputEnded:Connect(function()
    dragging = false
end)

SpeedSlider.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local pos = input.Position.X - SpeedSlider.AbsolutePosition.X
        local width = SpeedSlider.AbsoluteSize.X
        local percent = math.clamp(pos / width, 0, 1)
        local multiplier = percent * 3
        SpeedFill.Size = UDim2.new(percent, 0, 1, 0)
        SpeedLabel.Text = "Speed: " .. string.format("%.1fx", multiplier) .. "x"
        UpdateGlobalSpeed(multiplier)
    end
end)

-- Кнопки
StartBtn.MouseButton1Click:Connect(StartAutoStyle)
StopBtn.MouseButton1Click:Connect(StopAutoStyle)

FlickBtn.MouseButton1Click:Connect(function()
    for name, plr in pairs(SelectedTargets) do
        FlickPlayer(plr)
        task.wait(0.2)
    end
end)

-- Оновлення при зміні гравців
Players.PlayerAdded:Connect(RefreshPlayerList)
Players.PlayerRemoving:Connect(function(plr)
    SelectedTargets[plr.Name] = nil
    if Connections[plr.Name] then
        Connections[plr.Name]:Disconnect()
        Connections[plr.Name] = nil
    end
    RefreshPlayerList()
end)

-- Ініціалізація
RefreshPlayerList()
