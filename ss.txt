-- ========== ФУНКЦІЯ ФЛІКУ (телепорт до тебе) ==========
local function FlickPlayerToMe(target)
    if not target or not target.Character then return end
    
    local targetRoot = target.Character:FindFirstChild("Humanoid") and target.Character.Humanoid.RootPart
    local myChar = Player.Character
    local myRoot = myChar and myChar:FindFirstChild("Humanoid") and myChar.Humanoid.RootPart
    
    if targetRoot and myRoot then
        -- Телепортуємо гравця до мене
        targetRoot.CFrame = myRoot.CFrame * CFrame.new(0, 0, 2)
        targetRoot.Velocity = Vector3.new(0, 0, 0)
        
        -- Невелике затримання, щоб він не втік
        task.wait(0.1)
        targetRoot.CFrame = myRoot.CFrame * CFrame.new(0, 0, 2)
        
        print("Flicked " .. target.Name .. " to you!")
    end
end

-- ========== АВТОСТИЛ (механіка з першого скрипта) ==========
local function ApplyAutoStyle(target)
    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not AutoStyleEnabled or not target or not target.Character then
            if connection then connection:Disconnect() end
            return
        end
        
        local targetChar = target.Character
        local targetHumanoid = targetChar:FindFirstChildOfClass("Humanoid")
        local targetRoot = targetHumanoid and targetHumanoid.RootPart
        local myChar = Player.Character
        local myHumanoid = myChar and myChar:FindFirstChildOfClass("Humanoid")
        local myRoot = myHumanoid and myHumanoid.RootPart
        
        if targetRoot and myRoot then
            -- Слідуємо за гравцем
            targetRoot.CFrame = myRoot.CFrame * CFrame.new(0, 0, 2)
            targetRoot.Velocity = myRoot.Velocity
            
            -- Блокуємо сидіння
            if targetHumanoid then
                targetHumanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
            end
        end
    end)
    return connection
end

-- Запуск автостилу
local function StartAutoStyle()
    if not SelectedTarget then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Error",
            Text = "Select a target first!",
            Duration = 2
        })
        return
    end
    
    -- Зупиняємо старі підключення
    for _, conn in pairs(AutoStyleConnections) do
        if conn then conn:Disconnect() end
    end
    AutoStyleConnections = {}
    
    -- Запускаємо новий автостил
    AutoStyleConnections[1] = ApplyAutoStyle(SelectedTarget)
    AutoStyleEnabled = true
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "AutoStyle",
        Text = "Started on " .. SelectedTarget.Name,
        Duration = 2
    })
end

-- Зупинка автостилу
local function StopAutoStyle()
    for _, conn in pairs(AutoStyleConnections) do
        if conn then conn:Disconnect() end
    end
    AutoStyleConnections = {}
    AutoStyleEnabled = false
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "AutoStyle",
        Text = "Stopped",
        Duration = 2
    })
end

-- ========== ШВИДКІСТЬ ГРАВЦЯ ==========
local function SetPlayerSpeed(speed)
    PlayerSpeed = speed
    local char = Player.Character
    if char then
        local humanoid = char:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid.WalkSpeed = speed
        end
    end
end

-- ========== ТЕЛЕПОРТ ДО БАЗИ ==========
local function TeleportToBase()
    local success, result = pcall(function()
        local plotPos = workspace.Gardens and workspace.Gardens.Plot5 and workspace.Gardens.Plot5.PlotSizeReference
        if plotPos then
            Player.Character:PivotTo(plotPos.CFrame + Vector3.new(0, 100, 0))
            game:GetService("StarterGui"):SetCore("SendNotification", {
                Title = "TP",
                Text = "Teleported to base!",
                Duration = 2
            })
        else
            game:GetService("StarterGui"):SetCore("SendNotification", {
                Title = "Error",
                Text = "Base not found!",
                Duration = 2
            })
        end
    end)
    
    if not success then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Error",
            Text = "Cannot teleport right now",
            Duration = 2
        })
    end
end

-- Оновлення списку гравців
local function UpdatePlayerList()
    for _, child in pairs(PlayerListFrame:GetChildren()) do
        if child:IsA("Frame") then child:Destroy() end
    end
    
    local yPos = 5
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= Player then
            local btn = Instance.new("TextButton")
            btn.Size = UDim2.new(1, -10, 0, 30)
            btn.Position = UDim2.new(0, 5, 0, yPos)
            btn.BackgroundColor3 = Color3.fromRGB(55, 55, 60)
            btn.Text = plr.Name
            btn.TextColor3 = Color3.fromRGB(255, 255, 255)
            btn.Font = Enum.Font.Gotham
            btn.TextSize = 12
            btn.Parent = PlayerListFrame
            
            local btnCorner = Instance.new("UICorner")
            btnCorner.CornerRadius = UDim.new(0, 5)
            btnCorner.Parent = btn
            
            btn.MouseButton1Click:Connect(function()
                SelectedTarget = plr
                TargetDropdown.Text = plr.Name
                PlayerListFrame.Visible = false
                PlayerListFrame.Size = UDim2.new(1, -20, 0, 0)
            end)
            
            yPos = yPos + 35
        end
    end
    
    PlayerListFrame.CanvasSize = UDim2.new(0, 0, 0, yPos + 5)
end

-- Показати/сховати список

end)

-- Флік логіка (телепорт до тебе кожну секунду)
local FlickConnection = nil

local function StartFlick()
    if FlickConnection then FlickConnection:Disconnect() end
    
    FlickConnection = RunService.Heartbeat:Connect(function()
        if FlickEnabled and SelectedTarget then
            FlickPlayerToMe(SelectedTarget)
        end
    end)
end

local function StopFlick()
    if FlickConnection then
        FlickConnection:Disconnect()
        FlickConnection = nil
    end
end

FlickCheckbox.MouseButton1Click:Connect(function()
    if FlickEnabled then
        FlickEnabled = false
        StopFlick()
        FlickCheckmark.Visible = false
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Flick",
            Text = "Disabled",
            Duration = 1
        })
    else
        if not SelectedTarget then
            game:GetService("StarterGui"):SetCore("SendNotification", {
                Title = "Error",
                Text = "Select a target first!",
                Duration = 2
            })
            return
        end
        FlickEnabled = true
        StartFlick()
        FlickCheckmark.Visible = true
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Flick",
            Text = "Enabled - " .. SelectedTarget.Name .. " will teleport to you",
            Duration = 2
        })
    end
end)

-- ========== ПОВЗУНОК ШВИДКОСТІ ==========
local dragging = false
SpeedSlider.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
    end
end)

SpeedSlider.InputEnded:Connect(function()
    dragging = false
end)

SpeedSlider.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local pos = input.Position.X - SpeedSlider.AbsolutePosition.X
        local width = SpeedSlider.AbsoluteSize.X
        local percent = math.clamp(pos / width, 0, 1)
        local speedValue = 16 + (percent * 84) -- від 16 до 100
        SpeedFill.Size = UDim2.new(percent, 0, 1, 0)
        SpeedLabel.Text = "My Speed: " .. math.floor(speedValue)
        SetPlayerSpeed(speedValue)
    end
end)

-- ========== ОНОВЛЕННЯ ПРИ ЗМІНІ ГРАВЦІВ ==========
Players.PlayerRemoving:Connect(function(plr)
    if SelectedTarget == plr then
        SelectedTarget = nil
        TargetDropdown.Text = "Click to select player"
        if AutoStyleEnabled then StopAutoStyle() end
        if FlickEnabled then
            FlickEnabled = false
            StopFlick()
            FlickCheckmark.Visible = false
        end
        AutoStyleCheckmark.Visible = false
    end
end)
